Section 2: HTML to Remove

Implementation: Option A (Borrowing Calculator — Keep All 5 Steps)

The Purchase flow uses Option A and retains all existing steps including the situation, finances, and borrowing calculator.

Do NOT remove these steps. They are KEPT for the Purchase flow:

StepClass NameStatusPurpose
Step 2step_2 (situation)KEEPCollects situation data (marital status, first-time buyer); feeds calculateLoanCapacity() client-side
Step 3step_3 (finances)KEEPCollects financial data (income, deposit, max property value); feeds calculateLoanCapacity() client-side
Step 5step_5 (result with calculator)KEEPDisplays borrowing capacity calculation; exclusive to Purchase flow

Important: Situation & Finance Data

Situation (step_2) and Finance (step_3) data are NOT submitted to the backend. They are used client-side only to calculate the borrowing capacity in step_5. Only name, email, and phone are sent to the backend via Contact Form 7.

Section 3: Complete Replacement JavaScript

Overview

Replace your entire quiz JavaScript with this refactored version. It implements:

  • Flow-based step arrays for clean branching
  • Conditional step advancement based on answers.looking
  • Dead-end and redirect screen handlers
  • CTA handlers for "redirect to purchase", "go to Finance in Five", etc.
  • Flow-aware progress indicator updates
  • Simplified form submission with only relevant fields
// ============================================================
// HUNTER GALLOWAY QUIZ — REFACTORED JAVASCRIPT
// ============================================================
// Implementation: Flow-based step arrays (Approach B from spec)
// Handles branching logic for 5 different user paths

// Initialize quiz
(function() {
  // ---- CONFIGURATION ----
  // Define step sequences for each flow
  // IMPLEMENTATION: Option A - Purchase keeps borrowing calculator with full 5-step flow
  const FLOWS = {
    purchase: [1, 2, 3, 4, 5],  // Select → Situation → Finances → Contact → Borrowing Result (calculator)

    refinance: [1, 'refi_2', 4, 'refi_confirmation'],

    build: [1, 'dead-end-build'],
    commercial: [1, 'dead-end-commercial'],
    exploring: [1, 'dead-end-exploring'],
  };

  // ---- STATE ----
  let currentFlow = null;        // Current flow type (purchase, refinance, etc.)
  let flowStepIndex = 0;         // Current position within the flow array
  let answers = {
    looking: null,
    refiGoal: null,
    name: null,
    email: null,
    phone: null,
  };

  // ---- DOM REFERENCES ----
  let stepElements = {};         // Map of step ID to DOM element
  let mainButton, backButton, currentStepIndicator;

  // Initialize on page load
  function init() {
    // Build step element map
    const allSteps = document.querySelectorAll('.quiz_card .step');
    allSteps.forEach(step => {
      const classes = step.className.split(' ');
      const stepId = classes.find(c => c.startsWith('step_') || c === 'dead-end');
      if (stepId) {
        stepElements[stepId] = step;
      }
      // Also map by class for backwards compatibility
      classes.forEach(cls => {
        if (cls.startsWith('step_') || cls.startsWith('dead-end-')) {
          stepElements[cls] = step;
        }
      });
    });

    // Get button and indicator elements
    mainButton = document.querySelector('.quiz_card .button-next');
    backButton = document.querySelector('.quiz_card .button-back');
    currentStepIndicator = document.querySelector('.quiz_card .current-step');

    // Attach event listeners to Step 1 radio buttons
    document.querySelectorAll('.quiz_card input[name="looking"]').forEach(radio => {
      radio.addEventListener('change', (e) => {
        answers.looking = e.target.value;
      });
    });

    // Attach event listeners to Refinance goal radio buttons
    document.querySelectorAll('.quiz_card input[name="refiGoal"]').forEach(radio => {
      radio.addEventListener('change', (e) => {
        answers.refiGoal = e.target.value;
      });
    });

    // Show initial step (Step 1)
    showStep(1);
    updateButton();
  }

  // Main navigation: advance to next step in current flow
  function nextStep() {
    // Step 1: Determine flow based on answer
    if (flowStepIndex === 0) {
      const looking = answers.looking;

      if (!looking) {
        alert('Please select an option to continue.');
        return;
      }

      // Route based on selection
      if (!(looking in FLOWS)) {
        console.error('Unknown flow:', looking);
        return;
      }

      currentFlow = looking;
      console.log('Starting flow:', currentFlow);
    }

    // Advance within the current flow
    flowStepIndex++;
    const flowSteps = FLOWS[currentFlow];

    // Check if we've completed the flow
    if (flowStepIndex >= flowSteps.length) {
      console.log('Flow complete:', currentFlow);
      handleFlowComplete();
      return;
    }

    // Show next step in flow
    const nextStepId = flowSteps[flowStepIndex];
    showStep(nextStepId);
    updateButton();
  }

  // Navigate back to previous step in current flow
  function handleBackButton() {
    if (flowStepIndex <= 0) {
      // At the beginning of a flow; go back to Step 1
      resetQuiz();
      return;
    }

    // For dead-end screens, back button goes to Step 1
    const flowSteps = FLOWS[currentFlow];
    if (flowSteps[flowStepIndex].startsWith('dead-end-')) {
      resetQuiz();
      return;
    }

    flowStepIndex--;
    const prevStepId = flowSteps[flowStepIndex];
    showStep(prevStepId);
    updateButton();
  }

  // Show a specific step (by ID or numeric step number)
  function showStep(stepId) {
    // Hide all steps
    Object.values(stepElements).forEach(el => {
      if (el) el.classList.remove('active');
    });

    // Show the target step
    let stepEl = null;

    if (typeof stepId === 'number') {
      // Numeric step ID (Step 1, 4, etc.)
      stepEl = stepElements[`step_${stepId}`];
    } else {
      // String step ID (purchase_confirmation, refi_2, dead-end-build, etc.)
      // Try direct lookup first
      stepEl = stepElements[stepId];

      // Try with 'step_' prefix
      if (!stepEl) stepEl = stepElements[`step_${stepId}`];

      // Try with 'dead-end-' prefix if not found
      if (!stepEl) {
        const deadEndEl = document.querySelector(`.dead-end.dead-end-${stepId}`);
        if (deadEndEl) stepEl = deadEndEl;
      }
    }

    if (stepEl) {
      stepEl.classList.add('active');
    } else {
      console.warn('Step not found:', stepId);
    }

    updateProgressIndicator();
  }

  // Capture field data when advancing
  function captureFormData() {
    // Capture contact form fields (name, email, phone)
    const nameField = document.querySelector('.quiz_card input[name="name"]');
    const emailField = document.querySelector('.quiz_card input[name="email"]');
    const phoneField = document.querySelector('.quiz_card input[name="phone"]');

    if (nameField) answers.name = nameField.value;
    if (emailField) answers.email = emailField.value;
    if (phoneField) answers.phone = phoneField.value;
  }

  // Update button text and visibility based on flow position
  function updateButton() {
    const flowSteps = FLOWS[currentFlow];
    const isDeadEnd = flowSteps && flowSteps[flowStepIndex].startsWith('dead-end-');

    if (!mainButton || !backButton) return;

    // Determine button text
    if (isDeadEnd) {
      // Dead-end screens don't have a main next button
      mainButton.classList.add('hidden');
    } else if (flowStepIndex === flowSteps.length - 1) {
      // Last step in flow
      mainButton.textContent = 'Submit';
      mainButton.classList.remove('hidden');
    } else {
      // Middle steps
      mainButton.textContent = 'Continue';
      mainButton.classList.remove('hidden');
    }

    // Back button visibility
    if (flowStepIndex === 0 || isDeadEnd) {
      backButton.classList.add('hidden');
    } else {
      backButton.classList.remove('hidden');
    }
  }

  // Update progress indicator (step X of Y)
  function updateProgressIndicator() {
    if (!currentStepIndicator) return;

    const flowSteps = FLOWS[currentFlow];
    if (!flowSteps) return;

    // Count non-dead-end steps for display
    const displaySteps = flowSteps.filter(s => !s.startsWith('dead-end-'));
    const currentDisplayIndex = flowSteps
      .slice(0, flowStepIndex + 1)
      .filter(s => !s.startsWith('dead-end-')).length;

    currentStepIndicator.textContent = `${currentDisplayIndex} / ${displaySteps.length}`;
  }

  // Handle flow completion (submission or redirect)
  function handleFlowComplete() {
    captureFormData();

    switch (currentFlow) {
      case 'purchase':
        submitForm('purchase');
        break;
      case 'refinance':
        submitForm('refinance');
        break;
      case 'build':
      case 'commercial':
      case 'exploring':
        // These are handled by CTA buttons; shouldn't reach here
        console.log('Dead-end flow completed (button should handle navigation)');
        break;
    }
  }

  // Submit form with relevant fields only
  function submitForm(flowType) {
    const payload = {
      looking: answers.looking,
      name: answers.name,
      email: answers.email,
      phone: answers.phone,
    };

    if (flowType === 'refinance') {
      payload.refiGoal = answers.refiGoal;
    }

    console.log('Submitting form:', payload);

    // Find Contact Form 7 form and populate hidden fields
    const form = document.querySelector('.quiz_card form.wpcf7-form');
    if (form) {
      // Set form fields
      const inputs = {
        'quiz-looking': answers.looking,
        'quiz-name': answers.name,
        'quiz-email': answers.email,
        'quiz-phone': answers.phone,
        'quiz-refi-goal': answers.refiGoal || '',
      };

      Object.entries(inputs).forEach(([name, value]) => {
        const field = form.querySelector(`[name="${name}"]`);
        if (field) field.value = value;
      });

      // Submit the form
      form.submit();
    }
  }

  // Reset quiz to Step 1
  function resetQuiz() {
    currentFlow = null;
    flowStepIndex = 0;
    answers = {
      looking: null,
      refiGoal: null,
      name: null,
      email: null,
      phone: null,
    };

    // Clear all radio selections
    document.querySelectorAll('.quiz_card input[type="radio"]').forEach(radio => {
      radio.checked = false;
    });

    // Clear form fields
    document.querySelectorAll('.quiz_card input[type="text"], .quiz_card input[type="email"], .quiz_card input[type="tel"]').forEach(input => {
      input.value = '';
    });

    showStep(1);
    updateButton();
  }

  // ============================================================
  // CTA HANDLERS (for dead-end and redirect screens)
  // ============================================================

  window.handleBuildRedirectToPurchase = function() {
    answers.looking = 'purchase';
    currentFlow = 'purchase';
    flowStepIndex = 1;
    showStep(4); // Contact step
    updateButton();
  };

  window.handleCommercialRedirectToPurchase = function() {
    answers.looking = 'purchase';
    currentFlow = 'purchase';
    flowStepIndex = 1;
    showStep(4); // Contact step
    updateButton();
  };

  window.handleExploringToPurchase = function() {
    answers.looking = 'purchase';
    currentFlow = 'purchase';
    flowStepIndex = 1;
    showStep(4); // Contact step
    updateButton();
  };

  window.handleStartOver = function() {
    resetQuiz();
  };

  // ============================================================
  // PUBLIC API (attach to window for onclick handlers)
  // ============================================================

  window.quizNextStep = nextStep;
  window.quizHandleBack = handleBackButton;

  // Initialize when DOM is ready
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();

Important Notes on the JavaScript

  • FLOWS object: Defines step sequences for each path. Purchase uses [1, 2, 3, 4, 5] to keep the full 5-step flow with borrowing calculator. Other flows branch to dead-end or confirmation screens.
  • Step IDs: Can be numeric (1, 2, 3, 4, 5) or string IDs ('refi_2', 'refi_confirmation', 'dead-end-build', etc.).
  • CTA handlers: Called from onclick attributes on redirect screen buttons. They reset the flow and navigate to the appropriate step.
  • Form submission: Captures only name, email, phone (from Contact step). Situation and Finance data are used client-side only for the calculator; they are NOT submitted to the backend.
  • Backwards compatibility: Step elements are mapped by multiple naming patterns to work with existing HTML structure.
